import { NextApiRequest, NextApiResponse } from 'next';
import { Pool } from 'pg';

const pool = new Pool({
  connectionString: process.env.Database_URL, // Ensure the environment variable name is correct
  ssl: {
    rejectUnauthorized: false,
  },
});

export default async (req: NextApiRequest, res: NextApiResponse) => {
  if (req.method === 'POST') {
    const { username, password } = req.body;

    try {
      // Fetch user from the database
      const user = await pool.query('SELECT * FROM accounts WHERE username = $1', [username]);

      // Check if the user exists and compare the passwords
      if (user.rows.length === 0 || user.rows[0].password !== password) {
        return res.status(401).json({ message: 'Invalid username or password' });
      }

      // Return user ID and username for the context
      return res.status(200).json({
        message: 'Signed in successfully',
        id: user.rows[0].id, // Return the user ID
        username: user.rows[0].username // Return the username
      });
    } catch (error) {
      console.error('Error signing in:', error);
      return res.status(500).json({ error: 'Internal Server Error' });
    }
  } else {
    res.setHeader('Allow', ['POST']);
    res.status(405).end(`Method ${req.method} Not Allowed`);
  }
};
